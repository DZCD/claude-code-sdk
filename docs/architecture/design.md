## 1. Core Type System

### 1.1 Message Types

```typescript
// src/types/message.ts

/** Unique identifier for messages, tool calls, and sessions */
type Snowflake = string

/** Base message that all message types extend */
interface BaseMessage {
  /** Unique identifier */
  id: Snowflake
  /** ISO timestamp of when the message was created */
  createdAt: string
}

/** A message sent by the user */
interface UserMessage extends BaseMessage {
  role: 'user'
  content: string | ContentBlock[]
}

/** A message generated by the model */
interface AssistantMessage extends BaseMessage {
  role: 'assistant'
  content: string | ContentBlock[]
}

/** A tool result message (surfaced as user message to the API) */
interface ToolResultMessage extends BaseMessage {
  role: 'user'
  content: ToolResultBlock[]
}

/** A system prompt message */
interface SystemMessage {
  role: 'system'
  content: string
}

/** Union of all message types */
type Message = UserMessage | AssistantMessage | ToolResultMessage | SystemMessage

/** Content blocks within a message */
type ContentBlock =
  | TextBlock
  | ToolUseBlock
  | ToolResultBlock
  | ThinkingBlock

interface TextBlock {
  type: 'text'
  text: string
}

interface ToolUseBlock {
  type: 'tool_use'
  id: Snowflake
  name: string
  input: Record<string, unknown>
}

interface ToolResultBlock {
  type: 'tool_result'
  toolUseId: Snowflake
  content: string | ContentBlock[]
  isError?: boolean
}

interface ThinkingBlock {
  type: 'thinking'
  thinking: string
  signature?: string
}
```

### 1.2 Tool Type System

```typescript
// src/types/tool.ts

import type { z } from 'zod'

/** Input schema must be a Zod object */
type AnyZodObject = z.ZodType<Record<string, unknown>>

/** A runnable tool in the SDK */
interface Tool<Input extends AnyZodObject = AnyZodObject, Output = unknown> {
  /** Unique tool name used by the model */
  name: string
  
  /** One-line description for the model */
  description: string
  
  /** Zod schema for validating tool input */
  inputSchema: Input
  
  /** Execute the tool with validated input */
  execute(
    input: z.infer<Input>,
    context: ToolExecutionContext
  ): Promise<ToolResult<Output>>
  
  /** Whether this is a read-only operation (no side effects) */
  isReadOnly?(input: z.infer<Input>): boolean
  
  /** Whether this tool is safe to run concurrently */
  isConcurrencySafe?(): boolean
  
  /** Human-readable name for UI display */
  userFacingName?(input?: Partial<z.infer<Input>>): string
}

interface ToolExecutionContext {
  /** Abort signal for cancellation */
  signal: AbortSignal
  /** Logger instance */
  logger?: Logger
  /** Additional context from the session */
  [key: string]: unknown
}

interface ToolResult<Output = unknown> {
  /** Structured result data */
  data: Output
  /** Text content for the model to consume */
  content: string
  /** Whether this is an error result */
  isError?: boolean
  /** Additional messages to inject into the conversation */
  newMessages?: Message[]
}

/** Collection of tools */
type Tools = readonly Tool[]

/** Schema for tool definition in API calls */
interface ToolDefinition {
  name: string
  description: string
  input_schema: Record<string, unknown>
}
```

### 1.3 LLM Provider Types

```typescript
// src/llm/types.ts

/** Supported LLM providers */
type LLMProvider = 'anthropic' | 'bedrock' | 'vertex' | 'foundry'

/** Base configuration for all providers */
interface BaseLLMConfig {
  /** Model identifier (e.g., 'claude-sonnet-4-20250514') */
  model: string
  /** Maximum tokens to generate */
  maxTokens?: number
  /** Temperature for generation */
  temperature?: number
  /** Thinking/Extended thinking budget */
  thinkingBudget?: number
}

/** Anthropic Direct API configuration */
interface AnthropicConfig extends BaseLLMConfig {
  provider: 'anthropic'
  apiKey: string
  baseUrl?: string
}

/** AWS Bedrock configuration */
interface BedrockConfig extends BaseLLMConfig {
  provider: 'bedrock'
  region?: string
  accessKeyId?: string
  secretAccessKey?: string
  sessionToken?: string
}

/** Google Vertex AI configuration */
interface VertexConfig extends BaseLLMConfig {
  provider: 'vertex'
  projectId: string
  region?: string
  credentials?: string
}

/** Azure Foundry configuration */
interface FoundryConfig extends BaseLLMConfig {
  provider: 'foundry'
  resourceName: string
  apiKey?: string
  baseUrl?: string
}

type LLMConfig = AnthropicConfig | BedrockConfig | VertexConfig | FoundryConfig

/** Events emitted during streaming */
type StreamEvent =
  | { type: 'text'; text: string }
  | { type: 'tool_use_start'; id: Snowflake; name: string; input: Record<string, unknown> }
  | { type: 'tool_use_end'; id: Snowflake; output: ToolResultBlock }
  | { type: 'thinking'; thinking: string }
  | { type: 'error'; error: Error }
  | { type: 'done'; usage: TokenUsage }
  | { type: 'ping' }

interface TokenUsage {
  inputTokens: number
  outputTokens: number
  cacheCreationInputTokens?: number
  cacheReadInputTokens?: number
}

/** LLM connector interface */
interface LLMConnector {
  readonly provider: LLMProvider
  
  send(
    messages: Message[],
    tools: ToolDefinition[],
    options?: SendOptions
  ): AsyncIterable<StreamEvent>
  
  countTokens(messages: Message[]): Promise<number>
}

interface SendOptions {
  signal?: AbortSignal
  maxTokens?: number
  thinking?: { budgetTokens: number }
}
```

## 2. Module Designs

### 2.1 LLM Module (`src/llm/`)

**Responsibility**: Abstract over all LLM providers, provide unified streaming interface.

**Key Component: LLMClient**

```typescript
class LLMClient {
  constructor(config: LLMConfig)
  
  /** Send messages and get streaming response */
  stream(
    messages: Message[],
    tools?: ToolDefinition[],
    signal?: AbortSignal
  ): AsyncIterable<StreamEvent>
  
  /** Count tokens in messages */
  countTokens(messages: Message[]): Promise<number>
  
  /** Get the provider name */
  getProvider(): LLMProvider
}
```

**Provider Implementations**:
- `AnthropicConnector`: Uses `@anthropic-ai/sdk` Messages API
- `BedrockConnector`: Uses `@aws-sdk/client-bedrock-runtime` Converse API
- `VertexConnector`: Uses `google-auth-library` + Anthropic Vertex endpoint
- `FoundryConnector`: Uses Azure Foundry Anthropic endpoint

**Design Decisions**:
- Each provider is a standalone class implementing `LLMConnector`
- `LLMClient` is a facade that delegates to the appropriate connector
- Streaming uses AsyncIterable for compatibility with both `for await...of` and libraries like `it-pipe`

### 2.2 Tool Module (`src/tools/`)

**Responsibility**: Tool definition, registration, and execution infrastructure.

**Key Component: ToolRegistry**

```typescript
class ToolRegistry {
  /** Register one or more tools */
  register(...tools: Tool[]): void
  
  /** Get a tool by name */
  get(name: string): Tool | undefined
  
  /** Get all registered tools */
  getAll(): Tool[]
  
  /** Convert tools to API schema format */
  toAPISchemas(): ToolDefinition[]
  
  /** Execute a tool by name */
  execute(
    name: string,
    input: Record<string, unknown>,
    context: ToolExecutionContext
  ): Promise<ToolResult>
}
```

**Built-in Tool Implementations**:

| Tool | Class | Dependencies |
|------|-------|-------------|
| Bash | `BashTool` | node:child_process |
| FileRead | `FileReadTool` | node:fs |
| FileWrite | `FileWriteTool` | node:fs |
| FileEdit | `FileEditTool` | node:fs, diff |
| Glob | `GlobTool` | node:fs |
| Grep | `GrepTool` | node:fs |
| WebFetch | `WebFetchTool` | fetch API |
| WebSearch | `WebSearchTool` | fetch API |

**Design Decisions**:
- Tools are stateless: all state lives in the session/conversation manager
- Tool input is validated against a Zod schema before execution
- Each tool maps its result to both: (a) structured data for programmatic use, and (b) text content for the LLM

### 2.3 Conversation Module (`src/conversation/`)

**Responsibility**: Manage message history, tool calling loop, streaming responses.

**Key Component: ConversationManager**

```typescript
class ConversationManager {
  /** Add a user message and get a streaming response */
  send(
    message: string | Message,
    options?: SendOptions
  ): AsyncIterable<StreamEvent>
  
  /** Get conversation history */
  getHistory(): Message[]
  
  /** Reset conversation */
  reset(): void
  
  /** Get current token usage */
  getTokenUsage(): TokenUsage
  
  /** Manually trigger context compression */
  compact(): Promise<void>
}
```

**Tool Calling Loop** (`src/conversation/loop.ts`):

The conversation loop follows Claude Code's pattern:
1. Send messages to LLM
2. Stream response: collect text, tool_use blocks
3. If tool_use blocks: execute tools, inject results, continue loop
4. If no tool_use: return final assistant response
5. Check token budget and compact if needed

```typescript
async function* conversationLoop(
  llm: LLMClient,
  messages: Message[],
  tools: ToolRegistry,
  options: LoopOptions
): AsyncIterable<StreamEvent> {
  // See implementation section for details
}
```

**Design Decisions**:
- The conversation loop is a generator function for composability
- Auto-compact triggers based on token budget thresholds
- Maximum tool call depth prevents infinite loops (Claude Code uses a default of 50)

### 2.4 Session Module (`src/session/`)

**Responsibility**: High-level session management with persistence hooks.

**Key Component: SessionEngine**

```typescript
class SessionEngine {
  constructor(config: SDKConfig)
  
  /** Send a message and get a complete response (non-streaming) */
  async send(message: string): Promise<SessionResponse>
  
  /** Send a message and stream the response */
  stream(message: string): AsyncIterable<StreamEvent>
  
  /** Get or create a conversation */
  conversation(id?: string): ConversationManager
  
  /** Register custom tools */
  use(tool: Tool): void
  
  /** Load default tools */
  useDefaults(): void
  
  /** Set permission mode */
  withPermissionMode(mode: PermissionMode): void
  
  /** Get current config */
  getConfig(): SDKConfig
}

interface SessionResponse {
  content: string
  toolCalls: ToolCallRecord[]
  usage: TokenUsage
}

interface ToolCallRecord {
  toolName: string
  input: Record<string, unknown>
  output: unknown
  duration: number
}
```

**Design Decisions**:
- SessionEngine owns the LLM client, tool registry, and conversation manager
- Provides both streaming and non-streaming APIs
- Session is single-conversation (for multi-conversation, create multiple sessions)

### 2.6 Permission Module (`src/permission/`)

**Responsibility**: Control what tools can do based on mode and rules.

```typescript
class PermissionManager {
  constructor(mode: PermissionMode, rules?: PermissionRule[])
  
  /** Check if a tool call is permitted */
  async check(request: PermissionRequest): Promise<PermissionDecision>
  
  /** Set permission mode */
  setMode(mode: PermissionMode): void
  
  /** Add a permission rule */
  addRule(rule: PermissionRule): void
  
  /** Path validation for file operations */
  validatePath(path: string, allowedDirs: string[]): boolean
}
```

**Design Decisions**:
- `auto` mode: Always allow, but log decisions
- `manual` mode: Always require explicit approval (via callback)
- `plan` mode: Allow read-only tools, deny write tools
- Permission decision callback enables integration with authorization UIs

### 2.7 Config Module (`src/config/`)

**Responsibility**: Load, save, and merge configuration.

```typescript
class ConfigManager {

  merge(...configs: Partial<SDKConfig>[]): SDKConfig
  
  /** Save config to file */
  saveToFile(path: string, config: SDKConfig): Promise<void>

  getConfig(): SDKConfig
}
```

## 3. Public API Surface

### 3.1 Main Entry Point (builder pattern)

```typescript
// src/index.ts — Public API

import { ClaudeCodeSDK } from './session/engine'

// Primary instantiation
const sdk = await ClaudeCodeSDK.create({
  llm: {
    provider: 'anthropic',
    apiKey: process.env.ANTHROPIC_API_KEY,
    model: 'claude-sonnet-4-20250514',
  },
})

// Quick usage
const response = await sdk.send('Hello!')

// Streaming
const stream = sdk.stream('Write a poem')
for await (const event of stream) {
  if (event.type === 'text') process.stdout.write(event.text)
}
```
